439template <typename MainCameraTag>
440void RenderingSystem<MainCameraTag>::renderSSAO(float strength, float radius) {
441 SHKYERA_PROFILE("RenderingSystem<MainCameraTag>::renderSSAO");
442
443 static std::vector<glm::vec3> ssaoKernel;
444 static float usedRadius;
445
446 bool updatedKernel = false;
447 if (ssaoKernel.empty() || radius != usedRadius) {
448 constexpr int SSAOSamples = 128;
449 updatedKernel = true;
450 usedRadius = radius;
451
452 ssaoKernel.resize(SSAOSamples);
453 random::Uniform<float> sampler(0.0f, 1.0f);
454 for (int i = 0; i < SSAOSamples; ++i) {
455 glm::vec3 sample(
456 sampler() * 2.0 - 1.0,
457 sampler() * 2.0 - 1.0,
458 sampler() * 0.95 + 0.05);
459 sample = glm::normalize(sample);
460
461 float scale = (float)i / SSAOSamples;
462 scale = glm::mix(0.05f, usedRadius, scale * scale);
463 sample *= scale;
464
465 ssaoKernel[i] = sample;
466 }
467 }
468
469 const glm::mat4& projectionMatrix = _registry->getComponent<CameraComponent>(*_registry->getEntity<MainCameraTag>()).getProjectionMatrix();
470
471 UseShader ssaoShaderUsage(_ssaoShaderProgram);
472
473 _ssaoShaderProgram.setUniform("strength", strength);
474
475 _viewSpacePositionFrameBuffer.getTexture().activate(GL_TEXTURE0);
476 _ssaoShaderProgram.setUniform("depthTexture", 0);
477 _viewSpaceNormalFrameBuffer.getTexture().activate(GL_TEXTURE1);
478 _ssaoShaderProgram.setUniform("normalTexture", 1);
479
480 if (updatedKernel) {
481 for (int i = 0; i < 128; i++) {
482 _ssaoShaderProgram.setUniform("samples[" + std::to_string(i) + "]", ssaoKernel[i]);
483 }
484 }
485
486 // Set uniforms using the parameter pack
487 _ssaoShaderProgram.setUniform("projection", projectionMatrix);
488
489 // Draw fullscreen quad
490 _ssaoFrameBuffer.bind();
491 utils::drawFullscreenQuad();
492 _ssaoFrameBuffer.unbind();
493}

505 renderOutline(children);

510 // Drawing a silhouette

1036void RenderingSystem<MainCameraTag>::bloom(float threshold, float weight) {
1037 SHKYERA_PROFILE("RenderingSystem<MainCameraTag>::renderBloom");
1038
1039 glDisable(GL_DEPTH_TEST);
1040
1041 // Downscaling Pass
1042 utils::applyShaderToFrameBuffer(
1043 _downscaledFrameBuffers[0],
1044 _thresholdShaderProgram,
1045 {{"sceneTexture", &_mostRecentFrameBufferPtr->getTexture()}},
1046 utils::Uniform("threshold", threshold));
1047 for (size_t i = 1; i < BloomSteps; ++i) {
1048 utils::applyShaderToFrameBuffer(
1049 _downscaledFrameBuffers[i],
1050 _thresholdShaderProgram,
1051 {{"sceneTexture", &_downscaledFrameBuffers[i - 1].getTexture()}},
1052 utils::Uniform("threshold", 0.0f));
1053 }
1054
1055 // Horizontal Blur Pass
1056 for (size_t i = 0; i < BloomSteps; ++i) {
1057 utils::applyShaderToFrameBuffer(
1058 _horizontallyBluredDownscaledFrameBuffers[i],
1059 _horizontalBlurShaderProgram,
1060 {{"textureToBlur", &_downscaledFrameBuffers[i].getTexture()}},
1061 utils::Uniform("resolution", _downscaledFrameBuffers[i].getSize()));
1062 }
1063
1064 // Vertical Blur Pass
1065 for (size_t i = 0; i < BloomSteps; ++i) {
1066 utils::applyShaderToFrameBuffer(
1067 _fullyBluredDownscaledFrameBuffers[i],
1068 _verticalBlurShaderProgram,
1069 {{"textureToBlur", &_horizontallyBluredDownscaledFrameBuffers[i].getTexture()}},
1070 utils::Uniform("resolution", _horizontallyBluredDownscaledFrameBuffers[i].getSize()));
1071 }
1072
1073 // Blur Combination Pass
1074 auto* currentBuffer = &_fullyBluredDownscaledFrameBuffers[BloomSteps - 1];
1075 float currentWeight = 1.0f / (1 << (BloomSteps - 1));
1076
1077 for (int i = BloomSteps - 2; i >= 0; --i) {
1078 auto* nextBuffer = &_fullyBluredDownscaledFrameBuffers[i];
1079 float nextWeight = 1.0f / (1 << i);
1080
1081 utils::applyShaderToFrameBuffer(
1082 *currentBuffer,
1083 _weightedAdditionShaderProgram,
1084 {{"firstTexture", &currentBuffer->getTexture()},
1085 {"secondTexture", &nextBuffer->getTexture()}},
1086 utils::Uniform("firstWeight", currentWeight),
1087 utils::Uniform("secondWeight", nextWeight));
1088
1089 currentWeight += nextWeight;
1090 }
1091
1092 // Composite bloom with the original scene
1093 utils::applyShaderToFrameBuffer(
1094 _bloomedFrameBuffer,
1095 _weightedAdditionShaderProgram,
1096 {{"firstTexture", &_mostRecentFrameBufferPtr->getTexture()},
1097 {"secondTexture", &currentBuffer->getTexture()}},
1098 utils::Uniform("firstWeight", 1.0f),
1099 utils::Uniform("secondWeight", weight));
1100
1101 _mostRecentFrameBufferPtr = &_bloomedFrameBuffer;
1102}
